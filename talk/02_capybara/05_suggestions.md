# Use modules
## to create your own DSL

!SLIDE

    @@@ruby
    User.create(
      :username => 'jonas',
      :password => 'capybara')
    visit('/')
    fill_in('Username', :with => 'jonas')
    fill_in('Password', :with => 'capybara')
    click_button('Sign in')

!SLIDE

    @@@ruby
    sign_in_as('jonas')

!SLIDE

    @@@ruby
    module SignInHelpers
      def sign_in_as(name)
        User.create(
          :username => name,
          :password => 'capybara')
        visit('/')
        fill_in('Username', :with => name)
        fill_in('Password', :with => 'capybara')
        click_button('Sign in')
      end
    end

# Use Capybara selectors
## to describe parts of the page

!SLIDE

    @@@ruby
    find(:xpath, ".//li[contains(@class, 'todo')]" +
      "[contains(.,'#{name}')]")

!SLIDE

    @@@ruby
    find(:todo, name)

!SLIDE

    @@@ruby
    Capybara.add_selector :whisky do
      xpath do |name|
        ".//li[contains(@class, 'whisky')]" +
        "[contains(.,'#{name}')]"
      end
    end
